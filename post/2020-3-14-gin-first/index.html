Gin初体验

后端开发难免会接触到web,在Java的框架中spring一家独大，无论是web还是其他服务spring都是不二的选择；而golang的web方面选择特别的多，想gin,beego等，现在还处于不断造轮子的阶段，都在为开源做贡献，目前进行了一个Gin的初体验，总体上是让人很满意的，不管是gin还是beego都是以modules的这种形式，只允许加入自己需要的东西，减少负荷，以便加载到内存中的modules都是自己需要的。

特性都可以在gin的文档中寻找到，如果追求性能与高效，这是一个吸引人的亮点，在目前的开发中，都是最求极致的性能与高效。本来go已经在http中为我们做足了工作，建立起一个http请求及其简单。

package main

import (
	"fmt"
	"net/http"
	"time"
)

func greet(w http.ResponseWriter, r *http.Request) {
	fmt.Fprintf(w, "Hello World! %s", time.Now())
}

func header(w http.ResponseWriter, req *http.Request) {
	for name, headers := range req.Header {
		for _, h := range headers {
			fmt.Fprintf(w, "%v: %v\n", name, h)
		}
	}
}

func main() {
	http.HandleFunc("/", greet)
	http.HandleFunc("/headers", header)
	http.ListenAndServe(":8080", nil)
}

这样便建立起了一个http server,监听在8080端口。而gin建立起一个请求更为简单，一下为入门例子。

package main

import "github.com/gin-gonic/gin"

func main() {
	r := gin.Default()
	r.GET("/ping", func(c *gin.Context) {
		c.JSON(200, gin.H{
			"message": "pong",
		})
	})
	r.Run()
}

gin.Default()是一个默认的一个Engine,其中包含了Logger(),Recovery()模块，在gin中Logger称作为中间件，他会返回HandlerFunc,它是一个func(*Context)类型。

// Default returns an Engine instance with the Logger and Recovery middleware already attached.
func Default() *Engine {
	debugPrintWARNINGDefault()
	engine := New()
	engine.Use(Logger(), Recovery())
	return engine
}

当然这部分可以是可以自定义的，完全可以基于自己的需求去实现一个中间件，然后加入到gin的模块中去。

func LoggerFilter() gin.HandlerFunc {
	return func(c *gin.Context) {
		t := time.Now()
		// set  example variable
		c.Set("Demo", "hello")
		// before requests
		c.Next()
		// after requests
		latency := time.Since(t)
		log.Print(latency)

		// obtain sand  status
		status := c.Writer.Status()
		log.Println(status)
	}
}

func main() {
	r := gin.New()
	r.Use(LoggerFilter())
	r.GET("/test", func(c *gin.Context) {
		example := c.MustGet("example").(string)
		// print ："hello"
		log.Println(example)
	})

	r.Run(":8080")
}

自定义如上所示，new()是创建了一个没有任何中间件的一个Engine，再去加入自己需要的部分，或者是自定义的中间件。极大的赋予了开发者创作的权力。

从第一个的体验来说，我比较好奇的地方是可以运行多个服务的功能，这个功能目前还从未遇到过这种需求，为什么需要这种功能，看来得随着深入的使用之后才会知道其中的缘由。

package main

import (
	"github.com/gin-gonic/gin"
	"golang.org/x/sync/errgroup"
	"log"
	"net/http"
	"time"
)

var (
	g errgroup.Group
)

func RouterOne() http.Handler{
	one := gin.New()
	one.Use(gin.Recovery())
	one.GET("/", func(context *gin.Context) {
		context.JSON(
			http.StatusOK,
			gin.H{
				"code":http.StatusOK,
				"error":"welcome server one",
			},
			)
	})
	return one
}

func RouterTwo() http.Handler{
	two := gin.New()
	two.Use(gin.Recovery())
	two.GET("/", func(context *gin.Context) {
		context.JSON(
			http.StatusOK,
			gin.H{
				"code":http.StatusOK,
				"error":"welcome server two",
			},
		)
	})
	return two
}


func main() {
	serverone := &http.Server{
		Addr:              ":8080",
		Handler:           RouterOne(),
		TLSConfig:         nil,
		ReadTimeout:       5 * time.Second,
		ReadHeaderTimeout: 0,
		WriteTimeout:      5 * time.Second,
		IdleTimeout:       0,
		MaxHeaderBytes:    0,
	}
	servertwo := &http.Server{
		Addr:              ":8081",
		Handler:           RouterTwo(),
		TLSConfig:         nil,
		ReadTimeout:       5 * time.Second,
		ReadHeaderTimeout: 0,
		WriteTimeout:      5 * time.Second,
		IdleTimeout:       0,
		MaxHeaderBytes:    0,
	}

	g.Go(func() error {
		return serverone.ListenAndServe()
	})

	g.Go(func() error {
		return servertwo.ListenAndServe()
	})

	if err := g.Wait(); err != nil {
		log.Fatal(err)
	}
}

第一次体验，对于gin充满了好感，尝试着使用此web框架去完成一些小的demo,随着使用的深入，应该可以解答我对于多个服务的疑惑。
